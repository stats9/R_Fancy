get_pred_bayes <- function(x, Coef, dat){
    xx <- dat[x, ]
    beta0 <- Coef[1]
    beta_col2 <- 0
    beta_col3 <- Coef[2]
    beta_col4 <- Coef[3]
    beta_col5 <- Coef[4]
    beta_spine1 <- 0
    beta_spine2 <- Coef[5]
    beta_spine3 <- Coef[6]
    beta_width <- Coef[7]
    beta_weight <- Coef[8]
    color1 <- xx[1]
    spine1 <- xx[2]
    temp2 <-as.character(color1)
    beta_hat_col <- switch(temp2, 
    "2" = beta_col2, 
    "3" = beta_col3, 
    "4" = beta_col4, 
    beta_col5)
    temp4 <- as.character(spine1)
    beta_hat_spine <- switch(temp4, 
    "1" = beta_spine1, 
    "2" = beta_spine2, 
    beta_spine3)
    res <- exp(beta0 + beta_width * xx[3] + beta_weight * xx[4] + 
    beta_hat_col + beta_hat_spine) %>% setNames(NULL)
    return(res)
    }
