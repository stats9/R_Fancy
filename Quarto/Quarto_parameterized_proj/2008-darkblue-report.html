visualize

P <- filter_data |> 
        ggplot(aes(x = temperature, y = rainFall)) + 
        geom_point(color = params$color, size = 4, shape = 19) + 
        ggtitle(paste0("year: ", params$year)) + 
        theme_bw()

save figures

temp1 <- paste(paste(params$year, params$color, sep = "-"), "png", sep = ".")
